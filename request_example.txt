# Define the URL of your Flask app's predict endpoint
$url = "http://127.0.0.1:5000/predict"

# Create the JSON payload
$jsonPayload = @{
    matches = @(
        @{
            league = "I1"           # Replace with the appropriate league code
            HomeTeam = "Roma"       # Replace with the home team name
            AwayTeam = "Sassuolo"   # Replace with the away team name
            AvgH = 2.25             # Replace with the average home odds
            AvgD = 3.4              # Replace with the average draw odds
            AvgA = 3.10             # Replace with the average away odds
            AvgMORE25 = 1.72        # Replace with the average over 2.5 goals odds
            AvgCLESS25 = 2.10       # Replace with the average under 2.5 goals odds
        }
    )
} | ConvertTo-Json

# Send the POST request
$response = Invoke-RestMethod -Uri $url -Method Post -ContentType "application/json" -Body $jsonPayload

# Output the response
$response | ConvertTo-Json
